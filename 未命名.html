    function renderSubQuestions() {

        const container = document.getElementById("subQuestionsContainer");

            div.className = "sub-question";

            div.innerHTML = `

                <div class="sub-question-text">${sub.text}</div>

                <div class="rating" data-idx="${idx}">

                    ${[1,2,3,4,5].map(v => `<div class="rating-option" data-val="${v}">${v}</div>`).join('')}

                </div>

            `;

            // 绑定评分事件

            const options = div.querySelectorAll(".rating-option");

            options.forEach(opt => {

                opt.addEventListener("click", (e) => {

                    const val = parseInt(opt.getAttribute("data-val"));

                    currentSubAnswers[idx] = val - 1; // 存储0-4，方便后续转1-5

                    options.forEach(o => o.classList.remove("selected"));

                    opt.classList.add("selected");

                });

            });

            // 默认高亮第3个（3分）

            options[2].classList.add("selected");

            container.appendChild(div);

        });

    }

    }
